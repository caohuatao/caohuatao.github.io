Vue super flow

基于vue的流程图组件。

快速上手 →



# 当前版本

v1.3.5

# 安装


npm install vue-super-flow

yarn add vue-spuer-flow

# 全局引用


import SuperFlow from 'vue-super-flow'
import 'vue-super-flow/lib/index.css'

Vue.use(SuperFlow)

# 局部引用

<template>
  <super-flow></super-flow>
</template>

<script >
  import SuperFlow from 'vue-super-flow' 
  import 'vue-super-flow/lib/index.css'
  export default {
    components: {SuperFlow}   
  } 
</script>
